package comparable;

import java.util.Objects;

/**
 * The class contains details of the Book id ,name,author and price.
 * Books are sorted by implementing Comparable interface
 * and duplicates are removed using Overriding
 * hashCode and equals methods from object class
 * @author charles Student at RCA
 *
 * @see <a href ="https://docs.oracle.com/javase/8/docs/api/java/lang/Object.html">Object</a>
 * @see Object#equals(Object)
 */
public class Book implements Comparable {
    /**
     *id id of the book
     */
    private Integer id;
    /**
     *name name of the book
     */
    private String name;
    /**
     * author name of the author
     */
    private String author;
    /**
     * Book price of the book
     */
    private Integer price;

    /**
     * Default constructor
     */
    public Book() {
    }

    /**
     * Constructor
     * @param id id of the book
     * @param name name of the book
     * @param author  name of the person who wrote the book
     * @param price price of the book
     */

    public Book(Integer id, String name, String author,Integer price) {
        this.id = id;
        this.name = name;
        this.author = author;
        this.price=price;
    }

    /**
     *
     * @return the id of the book
     */

    public Integer getId() {
        return id;
    }

    public String getName() {
        return name;
    }

    public String getAuthor() {
        return author;
    }
    public Integer getPrice() {
        return price;
    }

    public void setPrice(Integer price) {
        this.price = price;
    }
    public String toString(){
        return STR."ID= \{this.id} name= \{this.name} author= \{this.author} price= \{this.price} ";
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Book book = (Book) o;
        return Objects.equals(id, book.id) && Objects.equals(name, book.name) && Objects.equals(author, book.author) && Objects.equals(price, book.price);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id, name, author, price);
    }

    @Override
    public int compareTo(Object o) {
        Book book = (Book) o;
        if (this.author.compareTo(book.author) != 0) {
            return this.author.compareTo(book.author);
        }
        if (this.name.compareTo(book.name) != 0) {
            return this.name.compareTo(book.name);
        }
        if (this.id.compareTo(book.id) != 0) {
            return this.id.compareTo(book.id);
        }
            return this.price.compareTo(book.price);
    }
}
